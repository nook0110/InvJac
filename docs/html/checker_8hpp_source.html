197 CheckResult::Result PerformCheck()
198 {
199 LOG_IF(FATAL, !map_) << "Attempting to perform a check without a map set.";
200 VLOG(4) << "Starting check with settings: iterations = "
201 << settings_.iterations << ", epsilon = " << settings_.epsilon
202 << ", check_contraction = " << settings_.check_contraction;
203
204 try
205 {
206 auto jacobian = map_->GetJacobian();
207 if (GiNaC::is_a<Complex>(jacobian))
208 {
209 VLOG(1) << "Jacobian is a complex number.";
210 return CheckResult::Keller{GiNaC::ex_to<Complex>(jacobian)};
211 }
212
213 for (size_t iteration = 0; iteration < settings_.iterations; ++iteration)
214 {
215 VLOG(2) << "Iteration " << iteration + 1 << " of "
216 << settings_.iterations;
217 static const std::array<Complex, 5> jacobian_values{
218 Complex(1), Complex(1, 2), Complex(-1), Complex(-1, 2), Complex(2)};
219
220 TestResult result;
221 Point point;
222 for (const auto& jacobian_value : jacobian_values)
223 {
224 VLOG(3) << "Testing with jacobian value: " << jacobian_value;
225 point = Point::GenerateRandom(map_->GetDimensions());
226 VLOG(3) << "Generated random point: " << point.ToStr();
227 point = map_->Image(
228 GeneratePointWithSpecificJacobian(*map_, point, jacobian_value)
229 .value_or(point));
230 VLOG(3) << "Mapped point: " << point.ToStr();
231
232 result = TestPoint(point);
233 if (!std::holds_alternative<SingularPoint>(result))
234 {
235 break;
236 }
237 }
238
239 if (std::holds_alternative<SingularPoint>(result))
240 {
241 VLOG(3) << "All points resulted in singular points, generating a new "
242 "random point.";
243 point = Point::GenerateRandom(map_->GetDimensions());
244 result = TestPoint(point);
245 }
246
247 if (std::holds_alternative<NonZeroJacobian>(result))
248 {
249 VLOG(2) << "Non-zero Jacobian found.";
250 if (map_->GetDimensions() == 2)
251 {
252 if (auto point = map_->HasContraction())
253 {
254 VLOG(2) << "Contraction found at point: " << point->ToStr();
255 return CheckResult::Contraction{*point};
256 }
257 }
258
259 VLOG(4) << "Non-zero Jacobian value: "
260 << std::get<NonZeroJacobian>(result).value;
261 return CheckResult::Failed{point,
262 std::get<NonZeroJacobian>(result).value};
263 }
264 }
265 }
266 catch (const std::exception& e)
267 {
268 LOG(ERROR) << "Exception occurred: " << e.what();
269 return CheckResult::Error{e.what()};
270 }
271 VLOG(1) << "Check passed after " << settings_.iterations << " iterations.";
272 return CheckResult::Passed{settings_.iterations};
273 }

295 TestResult TestPoint(const Point& point) const
296 {
297 VLOG(0) << "Testing point: " << point.ToStr();
298 auto solutions = Solver::GetInstance().Solve(*map_, point);
299
300 if (solutions.GetRootCount() < map_->GetExtensionDegree())
301 {
302 return SingularPoint{};
303 }
304
305 Complex sum_reciprocals = 0;
306
307 for (const auto& [multiplicity, solution] : solutions)
308 {
309 VLOG(3) << "Multiplicity: " << multiplicity
310 << " Solution: " << solution.ToStr();
311
312 auto jacobian = map_->GetJacobian();
313 auto term =
314 multiplicity / SubstituteJacobian(std::move(jacobian), solution);
315 sum_reciprocals += term;
316
317 VLOG(3) << "Term: " << term.evalf();
318 }
319
320 auto approximate_sum = GiNaC::abs(sum_reciprocals).to_double();
321 VLOG(1) << "Sum of reciprocals of Jacobians: " << approximate_sum;
322
323 bool result = approximate_sum < settings_.epsilon;
324 VLOG(1) << "Test result: " << (result ? "passed" : "failed");
325 if (result)
326 {
327 return ZeroJacobian{};
328 }
329 return NonZeroJacobian{approximate_sum};
330 }

355inline std::optional<Point> GeneratePointWithSpecificJacobian(
356 const Map& map, const Point& point, const Complex& value)
357{
358 VLOG(4) << "Generating point with specific Jacobian: " << value;
359 const auto dimensions = map.GetDimensions();
360
361 auto jacobian = map.GetJacobian();
362
363 std::vector<Polynomial> equations;
364
365 std::ranges::transform(
366 Symbols::GetSymbolsList(dimensions), point, std::back_inserter(equations),
367 [](const auto& symbol, const auto& coord) { return symbol - coord; });
368
369 for (size_t i = 0; i < dimensions; ++i)
370 {
371 Equation equation;
372 for (size_t j = 0; j < dimensions; ++j)
373 {
374 if (j == i)
375 {
376 equation.AppendEquation(jacobian - value);
377 }
378 else
379 {
380 equation.AppendEquation(equations[j]);
381 }
382 }
383 if (!equation.IsSquare())
384 {
385 continue;
386 }
387
388 auto solutions = equation.Solve();
389 auto solution = *solutions.begin();
390 DLOG_IF(FATAL, solution.point.GetDimensions() != dimensions)
391 << "The solution point does not have the expected dimensions.";
392 VLOG(1) << "Using point with jacobian equal to "
393 << ToStr(map.GetJacobian()
394 .subs(Symbols::GetSymbolsList(map.GetDimensions()),
395 solution.point.ToLst())
396 .evalf());
397 VLOG(1) << "Point is " << solution.point.ToStr();
398
399 return solution.point;
400 }
401 VLOG(1) << "Couldn't generate a point with jacobian equal to " << value;
402 return {};
