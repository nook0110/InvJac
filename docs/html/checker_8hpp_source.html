176 CheckResult::Result PerformCheck()
177 {
178 LOG_IF(FATAL, !map_) << "Attempting to perform a check without a map set.";
179
180 try
181 {
182 if (map_->GetDimensions() == 2)
183 {
184 if (auto point = map_->HasContraction())
185 {
186 return CheckResult::Contraction{*point};
187 }
188 }
189
190 for (size_t iteration = 0; iteration < settings_.iterations; ++iteration)
191 {
192 Point point = Point::GenerateRandom(map_->GetDimensions());
193 point = map_->Image(
194 GeneratePointWithUnitJacobian(*map_, point).value_or(point));
195
196 auto result = TestPoint(point);
197 if (result)
198 {
199 return CheckResult::Failed{point, *result};
200 }
201 }
202 }
203 catch (const std::exception& e)
204 {
205 LOG(ERROR) << "Exception occurred: " << e.what();
206 return CheckResult::Error{e.what()};
207 }
208 return CheckResult::Passed{settings_.iterations};
209 }
210
211 private:
217 std::optional<Complex> TestPoint(const Point& point) const
218 {
219 VLOG(0) << "Testing point: " << point.ToStr();
220 auto solutions = Solver::GetInstance().Solve(*map_, point);
221
222 Complex sum_reciprocals = 0;
223
224 for (const auto& [multiplicity, solution] : solutions)
225 {
226 auto jacobian = map_->GetJacobian();
227 auto term =
228 multiplicity / SubstituteJacobian(std::move(jacobian), solution);
229 sum_reciprocals += term;
230 }
231
232 auto approximate_sum = GiNaC::abs(sum_reciprocals).to_double();
233 VLOG(1) << "Sum of reciprocals of Jacobians: " << approximate_sum;
234
235 bool result = approximate_sum < settings_.epsilon;
236 VLOG(1) << "Test result: " << (result ? "passed" : "failed");
237 return result ? std::nullopt : std::optional<Complex>{approximate_sum};
238 }

263inline std::optional<Point> GeneratePointWithUnitJacobian(const Map& map,
264 const Point& point)
265{
266 const auto dimensions = map.GetDimensions();
267
268 auto jacobian = map.GetJacobian();
269
270 std::vector<Polynomial> equations;
271
272 std::ranges::transform(
273 Symbols::GetSymbolsList(dimensions), point, std::back_inserter(equations),
274 [](const auto& symbol, const auto& coord) { return symbol - coord; });
275
276 for (size_t i = 0; i < dimensions; ++i)
277 {
278 Equation equation;
279 for (size_t j = 0; j < dimensions; ++j)
280 {
281 if (j == i)
282 {
283 equation.AppendEquation(jacobian - 1);
284 }
285 else
286 {
287 equation.AppendEquation(equations[j]);
288 }
289 }
290 if (!equation.IsSquare())
291 {
292 continue;
293 }
294
295 auto solutions = equation.Solve();
296 auto solution = *solutions.begin();
297 DLOG_IF(FATAL, solution.point.GetDimensions() != dimensions)
298 << "The solution point does not have the expected dimensions.";
299 VLOG(1) << "Using point with jacobian equal to "
300 << ToStr(map.GetJacobian()
301 .subs(Symbols::GetSymbolsList(map.GetDimensions()),
302 solution.point.ToLst())
303 .evalf());
304 VLOG(1) << "Point is " << solution.point.ToStr();
305
306 return solution.point;
307 }
308 VLOG(1) << "Couldn't generate a point with jacobian equal to 1.";
309 return {};
